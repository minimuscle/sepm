import React, { Component } from 'react';
import { Popup, Button, Table, Icon, Divider, Header, Transition } from 'semantic-ui-react';
import '../style.css';
import api from '../../../api';
import Loadable from '../../Loadable';
 
const gantt = {
  backgroundColor: 'red',
  display: 'inline-block',
  color: 'white',
  height: '30px'
};
 
class Dashboard extends Component {
  constructor(props) {
    super(props);
    this.state = {
      hover: false,
      items: 3,
    }
  }
 
  handleAdd = () => {
    this.setState({items: this.state.items+1})
  }
 
  handleRemove = () => {
    this.setState({items: this.state.items-1})
  }
  
  handleMouseOver() {
    this.setState({
      hover: true
    })
  }
 
  handleMouseOut() {
    this.setState({
      hover: false
    })
  }
  
  taskView() {
    this.props.changeView("AddTask");
  }
 
  render() {
 
 
    return (
      <div className="content">
        {/*Gantt Chart here */}
        <div className="gantt">
        
          <Loadable promise={new api(this.props.user.token).getTasks()}>
            {(response, error, loading) => {
              if (loading) {
                return <div>loading</div>;
              } else {
                let maxDuration = Math.max(
                  ...response.map(task => task.duration)
                );
                return (
                  <div>
                    {/*<Button floated='right' icon labelPosition='left' primary size='small' onClick={this.taskView.bind(this)}>
                      <Icon name='plus' /> Add Task
                </Button><br/><br/>*/}
                    <Divider horizontal>
                      <Header as='h4'>
                        <Icon name='bar chart' />
                        Gantt Chart
                      </Header>
                    </Divider>
                    <Table celled striped>
                      
                      <Table.Header>
                        <Table.Row>
                          <Table.HeaderCell>Name</Table.HeaderCell>
                          <Table.HeaderCell>Gantt View</Table.HeaderCell>
                        </Table.Row>
                      </Table.Header>
 
                      <Table.Body>
                        {response.map((task, i) => (
                          <Table.Row key={i}>
                            <Table.Cell collapsing>{task.name}</Table.Cell>
                            <Table.Cell
                              onMouseOver={this.handleMouseOver.bind(this)}
                              onMouseOut={this.handleMouseOut.bind(this)}
                            >
                            <Popup trigger={
                              <div
                                style={{
                                  ...gantt,
                                  width:
                                    (task.duration / maxDuration) * 100 + '%',
                                }}
                              />
                            }
                            ><strong>{task.name}</strong>
                              <br/>
                              Duration: {task.duration} Days <br/>
                              Required Skills: {task.skills.join(', ')}
                              </Popup>
                            </Table.Cell>
                          </Table.Row>
                        ))}
                        </Table.Body>
                    </Table>
                  </div>
                );
              }
            }}
          </Loadable>
          <br/><br/>
        </div>
          {/*End Gantt Chart*/}
        
        {/*Tasks View*/}
        <Divider horizontal>
          <Header as='h4'>
            <Icon name='tasks' />
            Tasks
          </Header>
        </Divider>
        
 
        <Loadable promise={new api(this.props.user.token).getTasks()}>
            {(response, error, loading) => {
              if (loading) {
                return <div>loading</div>;
              } else {
                return (
                  <div>
                    Expand / Contract Tasks:&nbsp;&nbsp;
                    <Button.Group>
                      <Button disabled={this.state.items === 1} icon='minus' onClick={this.handleRemove} />
                      <Button disabled={this.state.items === response.length} icon='plus' onClick={this.handleAdd} />
                    </Button.Group>
                    <Table striped>
                      
                      <Table.Header fullWidth>
                        <Table.Row>
                          <Table.HeaderCell colSpan='6'>Tasks</Table.HeaderCell>
                        </Table.Row>
                      </Table.Header>
 
                      <Table.Body>
                      <Transition.Group as={Table.Row} duration={200}>
                      {response.map((task, i) => {
                        //This looks at how many items should be displayed.
                        if(i >= this.state.items) {
                          return null;
                        }
                        return (
 
                        <Table.Row key={i}>
                          <Table.Cell style={{padding: '20px 50px'}}>
                            <Popup trigger={
                              <Button circular basic color='green' icon='check' />
                            }
                            position="left center">Mark as Completed
                            </Popup>
                          </Table.Cell>
                          <Table.Cell>
                            ID: {task.id}<br/>
                            {task.name}
                          </Table.Cell>
                        </Table.Row>
                        
                        )
                      })}
                      </Transition.Group>
                      </Table.Body>
                    </Table>
                  </div>
                );
              }
            }}
        </Loadable>
        <br/><br/><br/>
      </div>
    );
  }
}
 
export default Dashboard;
