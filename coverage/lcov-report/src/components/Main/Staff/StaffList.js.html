import React, { Component } from 'react';
import 'components/Main/style.css';
import api from '../../../api';
import Loadable from '../../Loadable';
import { Table, TableRow } from 'semantic-ui-react';
 
const tableHeader = [
  "ID",
  "Name",
  "Type", // TODO name for this
  "Skills",
];
// TODO add column for allocated tasks
 
// TODO add styling for the table + new <div className="table-header"
class StaffList extends Component {
  render() {
    return (
      <div className="content">
        <Loadable promise={new api(this.props.token).getStaff()}>
          {(response, error, loading) => {
            if (loading) {
              return <div>loading</div>;
            } else {
              return (
                <div>
                  <Table striped>
                    <Table.Header>
                      <Table.Row>
                        {tableHeader.map((headers, i) => 
                         <Table.HeaderCell key={i}>{headers}</Table.HeaderCell>
                        )}
                      </Table.Row>
                    </Table.Header>
                    <Table.Body>
                      {response.map((data, j) => (
                        <Table.Row key={j}>
                          <Table.Cell>{data.id}</Table.Cell>
                          <Table.Cell>{data.name}</Table.Cell>
                          <Table.Cell>{data.staffType}</Table.Cell>
                          <Table.Cell>{data.skills.join(', ')}</Table.Cell>
                        </Table.Row>
                      ))}
                    </Table.Body>
                  </Table>
                </div>
              );
            }
          }}
        </Loadable>
      </div>
    );
  }
}
 
export default StaffList;
