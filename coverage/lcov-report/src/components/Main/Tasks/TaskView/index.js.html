import React, { Component } from 'react';
import '../../style.css'
import {Button, Table, Transition, Popup, Divider, Header, Icon} from 'semantic-ui-react'
import Loadable from '../../../Loadable'
import api from '../../../../api'
 
class TaskView extends Component {
  constructor(props) {
    super(props);
    this.state = {
      active: false
    }
  };
 
  complete(){
    if(this.state.active == true) {
      this.setState({active: false})
    } else {
      this.setState({active: true})
    }
    
    console.log(this.state.active);
  }
 
  addTask(){
    this.props.changeView("NewTask");
  }
 
  editTask(){
    this.props.changeView("EditTask");
  }
 
  render() {
    return (
      <div className="content">
        <Divider horizontal>
            <Header as='h2'>
            <Icon name='tasks' />
            Tasks
            </Header>
        </Divider>
        <Button floated="right" color="green" icon="plus" onClick={this.addTask.bind(this)}>Add New</Button>
        <div className="taskButton"></div>
        <Loadable promise={new api(this.props.user.token).getTasks()}>
            {(response, error, loading) => {
              if (loading) {
                return <div>loading</div>;
              } else {
                return (
                  <div>
                    
                    <Table striped>
                      <Table.Header fullWidth>
                        <Table.Row>
                          <Table.HeaderCell colSpan='6'>Tasks</Table.HeaderCell>
                        </Table.Row>
                      </Table.Header>
 
                      <Table.Body>
                      <Transition.Group as={Table.Row} duration={200}>
                      {response.map((task, i) => {
                        return(
                        <Table.Row key={i}>
                          <Table.Cell collapsing style={{padding: '20px 50px'}}>
                            <Popup trigger={
                            <Button toggle active={this.state.active} onClick={this.complete.bind(this)} circular icon='check' />
                            }
                            position="left center">Mark as Completed
                            </Popup>
                          </Table.Cell>
                          <Table.Cell>
                            <b>{task.name}</b>
                            <br/>
                            {task.description}
                          </Table.Cell>
                          <Table.Cell width="1" textAlign="right">
                            <Button icon="angle right" onClick={this.editTask.bind(this)}></Button>
                          </Table.Cell>
                        </Table.Row>
                        
                        )
                      })}
                      </Transition.Group>
                      </Table.Body>
                    </Table>
                  </div>
                );
              }
            }}
        </Loadable>
      </div>
    );
  }
}
 
export default TaskView;
