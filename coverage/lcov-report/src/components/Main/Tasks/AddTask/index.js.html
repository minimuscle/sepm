import React, { Component } from 'react';
import 'components/Main/style.css';
import _ from 'lodash';
import { Form, Button, Message, TextArea, Input, Label, Checkbox, Select } from 'semantic-ui-react';
import api from '../../../../api';
import Loadable from '../../../Loadable';
 
 
class AddTask extends Component {
  constructor(props) {
    super(props);
    this.state = {
      taskName: '',
      desc: '',
      dur: 0,
      isCrit: true,
      skills: [],
      staff: []
      // pull staff and skills from backend here, assign to variables, input into form
    };
 
    this.handleChange = this.handleChange.bind(this);
    this.handleSubmit = this.handleSubmit.bind(this);
    this.handleSelectChange = this.handleSelectChange.bind(this);
  }
 
 
  handleChange(event) {
    const target = event.target;
    const value = target.type === 'checkbox' ? target.checked : target.value;
    //maybe changed to a nested if or switch statement if need more inputs
    const name = target.name;
 
    this.setState({
      [name]: value
    });
  }
 
  handleSelectChange(event) {
    const name = event.target.name;
    this.setState({
      [name]: _.xor(this.state[name], [event.target.value])
    });
  }
 
  handleSubmit(event) {
    if(event)
      event.preventDefault(); //stops form submission default behaviour
      //convert form input into a json
      this.props.onSubmit({
        name: this.state.taskName
      });
  }
 
  back() {
    this.props.changeView("AddTask")
  }
 
  render() {
 
    const skills = [
      { key: 'r', text: 'React', value: 'r' },
      { key: 'js', text: 'Javascript', value: 'js' },
      { key: 'py', text: 'Python', value: 'py' }
    ]
 
    const staff = [
      { key: 'a', text: 'Alice', value: 'a' },
      { key: 'b', text: 'Bob', value: 'b' },
      { key: 'c', text: 'Charles', value: 'c' }
    ]
 
 
    return (
      <div className="content">
        <Button color="blue" onClick={this.back.bind(this)}>Back</Button>
        <Message attached><strong><h2>New Task</h2></strong><br/>Fill out the form below to create a new task</Message>
        <Form onSubmit={this.handleSubmit} className="attached fluid segment">
            <Form.Input 
            label="Name of Task"
            name="taskName"
            type="text"
            value={this.state.taskName}
            onChange={this.handleChange}
            placeholder="Fix render issues"
            width="4"/>
 
            <Form.Checkbox                   
              label="Critical Path?"
              name="isCrit"
              //checked={this.state.isCrit}
              onChange={this.handleChange}
            />  
 
            <Form.Field
            control={TextArea}
            label="Description"
            name="desc"
            value={this.state.desc}
            onChange={this.handleChange}
            width="8"
            />
 
            <Form.Input
              label="Duration"
              placeholder='4'
              width="2"
              name="dur"
              type="number"
              value={this.state.dur}
              onChange={this.handleChange}
            />
 
            <Form.Select 
            multiple
            label="Skills"
            name="skills"
            //value={this.state.skills}
            onChange={this.handleSelectChange}
            options={skills}
            width="4"
            />
 
            <Form.Select 
            multiple
            label="Staff"
            name="staff"
            //value={this.state.staff}
            onChange={this.handleSelectChange}
            options={staff}
            width="4"
            />
 
          
          <Button type="submit" value="Add Task" color="green">Add Task</Button> {/* submit form */}
          </Form>
      </div>
    );
  }
}
 
export default AddTask;
