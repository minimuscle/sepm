import React, { Component } from 'react';
import Login from 'containers/Login';
import './App.css';
import api from 'api';
import Main from '../../components/Main/Main';
 
class App extends Component {
  constructor(props) {
    super(props);
    this.state = {
      login: false,
      section: 'dashboard',
    };
    this.enableLogin = this.enableLogin.bind(this);
  }
 
  enableLogin = user => {
    this.setState({ login: true, user });
  };
 
  render() {
    if (this.state.login) {
      return <Main user={this.state.user} />;
    } else if (!this.state.login) {
      return <Login onLogin={this.enableLogin} login={api.login} />;
    }
  }
}
 
export default App;
