const BACKEND_ENDPOINT = process.env.REACT_APP_BACKEND_ENDPOINT;
 
// Main Api class, connects to backend
export default class API {
  constructor(token) {
    this.token = token;
  }
 
  static async login(username, password) {
    let result = await fetch(BACKEND_ENDPOINT + '/login', {
      method: 'POST',
      body: JSON.stringify({ username, password }),
    });
    if (!result.ok) {
      throw await result.json();
    }
    return await result.json();
  }
 
  async getTasks(username, password) {
    let result = await fetch(BACKEND_ENDPOINT + '/tasks');
    return await result.json();
  }
}
